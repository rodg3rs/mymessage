async function dispararMensagens() { const btn = document.getElementById('btn-envio-wa'); const status = document.getElementById('wa-status'); if (!confirm("Enviar 'Oi' de teste para o número configurado?")) return; btn.disabled = true; btn.innerText = "ENVIANDO..."; status.innerText = "Comunicando com o WhatsApp..."; try { const response = await fetch('/api/enviar-oi', { method: 'POST' }); const data = await response.json(); if(data.success) { status.innerHTML = "Oi enviado com sucesso!"; btn.innerText = "SUCESSO ✅"; } else { throw new Error(data.error); } } catch (e) { status.innerHTML = `Erro: ${e.message}`; btn.disabled = false; btn.innerText = "TENTAR NOVAMENTE"; } }
